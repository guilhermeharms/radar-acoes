<script>
  // ────────────────────────────────────────────────
  //  CONFIGURAÇÃO — preencha aqui
  // ────────────────────────────────────────────────
  const JSONBIN_KEY    = '$2a$10$ovXz1cJAkU6jtP7W7bQK8.SFUDWtfJ3QqTikoHpSov9URGfOObqY6';   // ex: $2a$10$abc...
  const JSONBIN_BIN_ID = '69bd90aac3097a1dd5438d1b';        // ex: 664a1f...
  // ────────────────────────────────────────────────

  const isConfigured = !JSONBIN_KEY.startsWith('SEU') && !JSONBIN_BIN_ID.startsWith('SEU');
  if (isConfigured) document.getElementById('configBanner').style.display = 'none';

  const BASE = `https://api.jsonbin.io/v3/b/${JSONBIN_BIN_ID}`;
  const HEADERS = { 'Content-Type': 'application/json', 'X-Master-Key': JSONBIN_KEY };

  const freqLabels = ['', 'Poucas menções', 'Várias menções', 'Urgente analisar'];
  const freqClass  = ['', 'badge-1', 'badge-2', 'badge-3'];

  let votedIds = JSON.parse(localStorage.getItem('radar_voted') || '[]');

  function switchTab(t) {
    document.querySelectorAll('.tab').forEach((el, i) =>
      el.classList.toggle('active', ['form', 'panel'][i] === t));
    document.getElementById('tab-form').style.display  = t === 'form'  ? 'block' : 'none';
    document.getElementById('tab-panel').style.display = t === 'panel' ? 'block' : 'none';
    if (t === 'panel') loadSuggestions();
  }

  async function readBin() {
    const r = await fetch(`${BASE}/latest`, { headers: { 'X-Master-Key': JSONBIN_KEY } });
    const d = await r.json();
    return Array.isArray(d.record?.sugestoes) ? d.record.sugestoes : [];
  }

  async function writeBin(list) {
    await fetch(BASE, { method: 'PUT', headers: HEADERS, body: JSON.stringify({ sugestoes: list }) });
  }

  function showToast(msg, type) {
    const el = document.getElementById('toast');
    el.textContent = msg;
    el.className = 'toast ' + type;
    setTimeout(() => { el.className = 'toast'; }, 3500);
  }

  async function submitSuggestion() {
    if (!isConfigured) { showToast('Configure JSONBIN_KEY e JSONBIN_BIN_ID no código.', 'err'); return; }
    const ticker   = document.getElementById('ticker').value.trim().toUpperCase();
    const assessor = document.getElementById('assessor').value.trim();
    const descricao = document.getElementById('descricao').value.trim();
    const freq     = parseInt(document.getElementById('freq').value);
    if (!ticker || !assessor || !descricao) { showToast('Preencha todos os campos.', 'err'); return; }

    const btn = document.getElementById('submitBtn');
    btn.disabled = true; btn.textContent = 'Enviando…';

    try {
      const list = await readBin();
      const entry = {
        id: Date.now().toString(36),
        ticker, assessor, descricao, freq, votes: 0,
        date: new Date().toLocaleDateString('pt-BR')
      };
      list.unshift(entry);
      await writeBin(list);
      document.getElementById('ticker').value = '';
      document.getElementById('assessor').value = '';
      document.getElementById('descricao').value = '';
      showToast('Sugestão enviada! Seus colegas já podem ver.', 'ok');
    } catch (e) {
      showToast('Erro ao salvar. Verifique as credenciais.', 'err');
    }

    btn.disabled = false; btn.textContent = 'Enviar sugestão';
  }

  async function upvote(id) {
    if (votedIds.includes(id)) return;
    try {
      const list = await readBin();
      const item = list.find(x => x.id === id);
      if (item) item.votes = (item.votes || 0) + 1;
      await writeBin(list);
      votedIds.push(id);
      localStorage.setItem('radar_voted', JSON.stringify(votedIds));
      loadSuggestions();
    } catch(e) {}
  }

  async function loadSuggestions() {
    if (!isConfigured) {
      document.getElementById('cards').innerHTML = '<div class="empty">Configure as credenciais do JSONBin para ver as sugestões.</div>';
      return;
    }
    document.getElementById('cards').innerHTML = '<div class="loading">Carregando…</div>';
    try {
      const list = await readBin();
      const urgent = list.filter(x => x.freq === 3).length;
      const ativos = new Set(list.map(x => x.ticker)).size;

      document.getElementById('stats').innerHTML = `
        <div class="stat"><div class="stat-val">${list.length}</div><div class="stat-lbl">Sugestões</div></div>
        <div class="stat"><div class="stat-val">${ativos}</div><div class="stat-lbl">Ativos únicos</div></div>
        <div class="stat"><div class="stat-val">${urgent}</div><div class="stat-lbl">Urgentes</div></div>`;

      document.getElementById('panelCount').textContent =
        list.length === 0 ? 'Nenhuma sugestão ainda' : `${list.length} sugestão${list.length !== 1 ? 's' : ''}`;

      if (list.length === 0) {
        document.getElementById('cards').innerHTML = '<div class="empty">Nenhuma sugestão ainda. Seja o primeiro!</div>';
        return;
      }

      const sorted = [...list].sort((a, b) => (b.votes || 0) - (a.votes || 0));
      document.getElementById('cards').innerHTML = sorted.map(s => `
        <div class="card">
          <div class="card-top">
            <span class="ticker">${s.ticker}</span>
            <span class="badge ${freqClass[s.freq]}">${freqLabels[s.freq]}</span>
          </div>
          <div class="card-desc">${s.descricao}</div>
          <div class="card-footer">
            <span class="card-meta">${s.assessor} · ${s.date || ''}</span>
            <button class="btn-upvote${votedIds.includes(s.id) ? ' voted' : ''}" onclick="upvote('${s.id}')">
              ▲ ${s.votes || 0} apoios
            </button>
          </div>
        </div>`).join('');
    } catch(e) {
      document.getElementById('cards').innerHTML = '<div class="empty">Erro ao carregar. Verifique as credenciais.</div>';
    }
  }
</script>

   Cole em: JSONBIN_BIN_ID = 'cole aqui'

4. Suba o arquivo no Vercel:
   - Acesse https://vercel.com → New Project → "Deploy from file"
   - Arraste este HTML → clique Deploy
   - Vercel gera um link público (ex: radar-assessores.vercel.app)

   OU GitHub Pages:
   - Crie um repo público → suba o arquivo como index.html
   - Settings → Pages → Source: main → /root
